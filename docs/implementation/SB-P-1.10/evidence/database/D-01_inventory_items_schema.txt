                                 Table "public.inventory_items"
   Column    |           Type           | Collation | Nullable |             Default             
-------------+--------------------------+-----------+----------+---------------------------------
 id          | uuid                     |           | not null | gen_random_uuid()
 business_id | uuid                     |           | not null | 
 name        | text                     |           | not null | 
 base_unit   | text                     |           | not null | 
 status      | inventory_item_status    |           | not null | 'active'::inventory_item_status
 created_by  | uuid                     |           | not null | 
 created_at  | timestamp with time zone |           | not null | now()
 updated_at  | timestamp with time zone |           | not null | now()
Indexes:
    "inventory_items_pkey" PRIMARY KEY, btree (id)
    "inventory_items_business_name_uniq" UNIQUE CONSTRAINT, btree (business_id, name)
    "inventory_items_business_status_idx" btree (business_id, status, name)
    "inventory_items_id_business_uniq" UNIQUE CONSTRAINT, btree (id, business_id)
Check constraints:
    "inventory_items_base_unit_check" CHECK (length(btrim(base_unit)) > 0)
    "inventory_items_name_check" CHECK (length(btrim(name)) > 0)
Foreign-key constraints:
    "inventory_items_business_id_fkey" FOREIGN KEY (business_id) REFERENCES businesses(id) ON DELETE CASCADE
Referenced by:
    TABLE "inventory_movements" CONSTRAINT "inventory_movements_item_business_fk" FOREIGN KEY (item_id, business_id) REFERENCES inventory_items(id, business_id) ON DELETE RESTRICT
Policies:
    POLICY "Owners can insert their inventory items" FOR INSERT
      TO authenticated
      WITH CHECK (((business_id IN ( SELECT businesses.id
   FROM businesses
  WHERE (businesses.owner_id = auth.uid()))) AND (created_by = auth.uid())))
    POLICY "Owners can update their inventory items" FOR UPDATE
      TO authenticated
      USING ((business_id IN ( SELECT businesses.id
   FROM businesses
  WHERE (businesses.owner_id = auth.uid()))))
      WITH CHECK ((business_id IN ( SELECT businesses.id
   FROM businesses
  WHERE (businesses.owner_id = auth.uid()))))
    POLICY "Owners can view their inventory items" FOR SELECT
      TO authenticated
      USING ((business_id IN ( SELECT businesses.id
   FROM businesses
  WHERE (businesses.owner_id = auth.uid()))))
Triggers:
    inventory_items_guard_trg BEFORE UPDATE ON inventory_items FOR EACH ROW EXECUTE FUNCTION inventory_items_guard()
    update_inventory_items_updated_at BEFORE UPDATE ON inventory_items FOR EACH ROW EXECUTE FUNCTION update_updated_at_column()
